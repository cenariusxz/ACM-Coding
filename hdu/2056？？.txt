#include<stdio.h>

int main()
{
	double a[7],b[7],c[7],d[7];
	while (scanf("%lf%lf%lf%lf%lf%lf%lf%lf",&a[5],&b[5],&a[6],&b[6],&c[5],&d[5],&c[6],&d[6])!=EOF)
	{
		int i,j,count=0;
		double s,t,x[5],y[5];
		{																				//排点，分别以a1b1、a3b3；c1d1、c3d3为副对角线
			a[4]=a[1]=(a[5]<a[6])?a[5]:a[6];
			a[2]=a[3]=(a[5]>a[6])?a[5]:a[6];
			c[4]=c[1]=(c[5]<c[6])?c[5]:c[6];
			c[2]=c[3]=(c[5]>c[6])?c[5]:c[6];
			b[2]=b[1]=(b[5]<b[6])?b[5]:b[6];
			b[4]=b[3]=(b[5]>b[6])?b[5]:b[6];
			d[2]=d[1]=(d[5]<d[6])?d[5]:d[6];
			d[4]=d[3]=(d[5]>d[6])?d[5]:d[6];
		}
		x[1]=a[1];x[2]=a[2];x[3]=c[1];x[4]=c[2];										//x数组为四个横坐标，y数组为四个纵坐标
		y[1]=b[1];y[2]=b[4];y[3]=d[1];y[4]=d[4];
		for (i=1;i<5;i++) if (a[i]>=c[1]&&a[i]<=c[3]&&b[i]>=d[1]&&b[i]<=d[3]) count++;	//判断第一个三角形是否有点在第二个三角形内
		for (i=1;i<5;i++) if (c[i]>=a[1]&&c[i]<=a[3]&&d[i]>=b[1]&&d[i]<=b[3]) count++;	//判断第二个三角形是否有点在第一个三角形内
		if (count==0) s=0;																//互相没有交叉点，面积为0
		else 
		{
			for (i=1;i<=4;i++)															//将横纵坐标排序
			{
				for (j=i+1;j<=4;j++)
				{
					if (x[i]>x[j])
					{
						t=x[i];
						x[i]=x[j];
						x[j]=t;
					}
					if (y[i]>y[j])
					{
						t=y[i];
						y[i]=y[j];
						y[j]=t;
					}
				}
			}
			s=(x[3]-x[2])*(y[3]-y[2]);													//面积为中间四个点构成的面积
		}
		printf("%.2lf\n",s);
	}
	return 0;
}
