#include<stdio.h>

int main()
{
	double a[7],b[7],c[7],d[7];
	while (scanf("%lf%lf%lf%lf%lf%lf%lf%lf",&a[5],&b[5],&a[6],&b[6],&c[5],&d[5],&c[6],&d[6])!=EOF)
	{
		int i,n,count=0;
		double x[5],y[5],s;
		{
			a[1]=(a[5]<a[6])?a[5]:a[6];
			a[3]=(a[5]>a[6])?a[5]:a[6];
			a[4]=a[1];a[2]=a[3];
			c[1]=(c[5]<c[6])?c[5]:c[6];
			c[3]=(c[5]>c[6])?c[5]:c[6];
			c[4]=c[1];c[2]=c[3];
			b[1]=(b[5]<b[6])?b[5]:b[6];
			b[3]=(b[5]>b[6])?b[5]:b[6];
			b[2]=b[1];b[4]=b[3];
			d[1]=(d[5]<d[6])?d[5]:d[6];
			d[3]=(d[5]>d[6])?d[5]:d[6];
			d[2]=d[1];d[4]=d[3];
		}
		for (i=1;i<5;i++)
		{
			if (a[i]>=c[1]&&a[i]<=c[3]&&b[i]>=d[1]&&b[i]<=d[3])
			{
				x[i]=a[i];
				y[i]=b[i];
				count++;
				n=i;
			}
		}
		if (count==4) s=(x[2]-x[1])*(y[3]-y[2]);
		if (count==2) 
		{
			if (a[1]>=c[1]&&a[1]<=c[2]&&b[1]>=d[2]&&b[4]<=d[4]) s=(b[4]-b[1])*(c[2]-a[1]);
			else
				if (a[4]>=c[1]&&a[3]<=c[2]&&b[4]>=d[1]&&b[4]<=d[4]) s=(a[3]-a[4])*(b[4]-d[1]);
				else
					if (a[2]>=c[1]&&a[2]<=c[2]&&b[2]>=d[1]&&b[3]<=d[4]) s=(b[3]-b[2])*(a[2]-c[1]);
					else s=(a[2]-a[1])*(d[3]-b[1]);
		}
		if (count==1)
		{
			if (n==1) s=(c[3]-a[1])*(d[3]-b[1]);
			else
				if (n==2) s=(a[2]-c[4])*(d[4]-b[2]);
				else
					if (n==3) s=(a[3]-c[1])*(b[3]-d[1]);
					else s=(a[4]-c[2])*(d[2]-b[4]);
		}
		if (count==0)
		{
			for (i=1;i<5;i++)
			{
				if (c[i]>=a[1]&&c[i]<=a[3]&&d[i]>=b[1]&&d[i]<=b[3])
				{
					x[i]=c[i];
					y[i]=d[i];
					count++;
					n=i;
				}
			}
			if (count==4) s=(x[2]-x[1])*(y[3]-y[2]);
			if (count==2) 
			{
				if (c[1]>=a[1]&&c[1]<=a[2]&&d[1]>=b[2]&&d[4]<=b[4]) s=(d[4]-d[1])*(a[2]-c[1]);
				else
					if (c[4]>=a[1]&&c[3]<=a[2]&&d[4]>=b[1]&&d[4]<=b[4]) s=(c[3]-c[4])*(d[4]-b[1]);
					else
						if (c[2]>=a[1]&&c[2]<=a[2]&&d[2]>=b[1]&&d[3]<=b[4]) s=(d[3]-d[2])*(c[2]-a[1]);
						else s=(c[2]-c[1])*(b[3]-d[1]);
			}
			else s=0.00;
		}
		printf("%.2lf\n",s);
	}
	return 0;
}
